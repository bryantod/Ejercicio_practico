#include <stdio.h>
#include <stdbool.h>
#include <math.h>

bool es_primo(int numero) {
    if (numero <= 1) return false;
    if (numero == 2) return true;
    if (numero % 2 == 0) return false;
    
    for (int i = 3; i <= sqrt(numero); i += 2) {
        if (numero % i == 0) {
            return false;
        }
    }
    return true;
}

void mostrar_primos(int cantidad) {
    printf("Los primeros %d numeros primos son:\n", cantidad);
    
    int encontrados = 0;
    int numero = 2;
    
    while (encontrados < cantidad) {
        if (es_primo(numero)) {
            printf("%d ", numero);
            encontrados++;
        }
        numero++;
    }
    printf("\n");
}

int main() {
    int opcion, numero, cantidad;
    
    printf("=== CALCULADORA DE NUMEROS PRIMOS ===\n");
    
    do {
        printf("\nMenu:\n");
        printf("1. Verificar si un numero es primo\n");
        printf("2. Mostrar primeros N numeros primos\n");
        printf("3. Salir\n");
        printf("Seleccione una opcion: ");
        scanf("%d", &opcion);
        
        switch(opcion) {
            case 1:
                printf("Ingrese un numero: ");
                scanf("%d", &numero);
                
                if (es_primo(numero)) {
                    printf("%d ES un numero primo.\n", numero);
                } else {
                    printf("%d NO es un numero primo.\n", numero);
                }
                break;
                
            case 2:
                printf("Cuantos numeros primos desea ver? ");
                scanf("%d", &cantidad);
                
                if (cantidad > 0) {
                    mostrar_primos(cantidad);
                } else {
                    printf("Por favor ingrese un numero positivo.\n");
                }
                break;
                
            case 3:
                printf("Saliendo...\n");
                break;
                
            default:
                printf("Opcion no valida. Intente de nuevo.\n");
        }
        
    } while(opcion != 3);
    
    return 0;
}
